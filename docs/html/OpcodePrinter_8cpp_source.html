9std::string OpcodePrinter::CodeToString(u1 _code[], const u4 code_lenght) {
10 StringBuffer = "";
11 code_iterator = 0;
12 code = _code;
13 while(code_iterator != code_lenght){
14 StringBuffer.append(" ");
15 std::ostringstream oss;
16 oss << std::setw(3) << code_iterator << ":";
17 std::string formatedIndex = oss.str();
18 StringBuffer.append(formatedIndex);
19 u1 bytecode = code[code_iterator++];
20 (this->*bytecodeFuncs[bytecode])();
21 }
22 return StringBuffer;
23}

677void OpcodePrinter::tableswitch() {
678 std::string buffer;
679 size_t start_index = code_iterator;
680 u1 index = code[code_iterator++]; // Índice do opcode
681
682 // Verifica se há bytes suficientes para ajustar o padding e ler default, low e high (12 bytes no total)
683 size_t padding_end = (code_iterator + 3) & ~3;
684 if (padding_end + 12 > code_size) {
685 std::cerr << "Erro: Acesso fora dos limites do array 'code' em table.A tester aguento nada" << std::endl;
686 return;
687 }
688
689 // Ajusta o padding
690 code_iterator = padding_end;
691
692 // Lendo default (4 bytes)
693 int defaultoffset = (code[code_iterator] << 24) |
694 (code[code_iterator + 1] << 16) |
695 (code[code_iterator + 2] << 8) |
696 code[code_iterator + 3];
697 code_iterator += 4;
698
699 // Lendo low (4 bytes)
